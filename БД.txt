CREATE TABLE warehouses (
    warehouse_id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    address TEXT NOT NULL,
    description TEXT
);
CREATE TABLE storage_zones (
    zone_id SERIAL PRIMARY KEY,
    warehouse_id INT NOT NULL,
    zone_code VARCHAR(50) NOT NULL,
    description TEXT,

    CONSTRAINT fk_zone_warehouse
        FOREIGN KEY (warehouse_id)
        REFERENCES warehouses (warehouse_id)
        ON DELETE CASCADE
);
CREATE TABLE product_categories (
    category_id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description TEXT
);
CREATE TABLE products (
    product_id SERIAL PRIMARY KEY,
    name VARCHAR(150) NOT NULL,
    category_id INT NOT NULL,
    article VARCHAR(50) UNIQUE NOT NULL,
    unit VARCHAR(20) NOT NULL,
    min_stock INT DEFAULT 0 CHECK (min_stock >= 0),

    CONSTRAINT fk_product_category
        FOREIGN KEY (category_id)
        REFERENCES product_categories (category_id)
);
CREATE TABLE suppliers (
    supplier_id SERIAL PRIMARY KEY,
    name VARCHAR(150) NOT NULL,
    contact_person VARCHAR(100),
    phone VARCHAR(30),
    email VARCHAR(100)
);
CREATE TABLE deliveries (
    delivery_id SERIAL PRIMARY KEY,
    supplier_id INT NOT NULL,
    warehouse_id INT NOT NULL,
    delivery_date DATE NOT NULL,
    status VARCHAR(30) NOT NULL,

    CONSTRAINT fk_delivery_supplier
        FOREIGN KEY (supplier_id)
        REFERENCES suppliers (supplier_id),

    CONSTRAINT fk_delivery_warehouse
        FOREIGN KEY (warehouse_id)
        REFERENCES warehouses (warehouse_id)
);
CREATE TABLE delivery_items (
    delivery_item_id SERIAL PRIMARY KEY,
    delivery_id INT NOT NULL,
    product_id INT NOT NULL,
    quantity INT NOT NULL CHECK (quantity > 0),
    purchase_price NUMERIC(10,2) NOT NULL CHECK (purchase_price >= 0),

    CONSTRAINT fk_item_delivery
        FOREIGN KEY (delivery_id)
        REFERENCES deliveries (delivery_id)
        ON DELETE CASCADE,

    CONSTRAINT fk_item_product
        FOREIGN KEY (product_id)
        REFERENCES products (product_id)
);
CREATE TABLE stock_balances (
    product_id INT NOT NULL,
    warehouse_id INT NOT NULL,
    zone_id INT NOT NULL,
    quantity INT NOT NULL DEFAULT 0 CHECK (quantity >= 0),

    PRIMARY KEY (product_id, warehouse_id, zone_id),

    CONSTRAINT fk_stock_product
        FOREIGN KEY (product_id)
        REFERENCES products (product_id),

    CONSTRAINT fk_stock_warehouse
        FOREIGN KEY (warehouse_id)
        REFERENCES warehouses (warehouse_id),

    CONSTRAINT fk_stock_zone
        FOREIGN KEY (zone_id)
        REFERENCES storage_zones (zone_id)
);
CREATE TABLE shipments (
    shipment_id SERIAL PRIMARY KEY,
    warehouse_id INT NOT NULL,
    shipment_date DATE NOT NULL,
    recipient VARCHAR(150) NOT NULL,
    status VARCHAR(30) NOT NULL,

    CONSTRAINT fk_shipment_warehouse
        FOREIGN KEY (warehouse_id)
        REFERENCES warehouses (warehouse_id)
);
CREATE TABLE employees (
    employee_id SERIAL PRIMARY KEY,
    full_name VARCHAR(150) NOT NULL,
    position VARCHAR(100) NOT NULL,
    phone VARCHAR(30),
    email VARCHAR(100)
);
